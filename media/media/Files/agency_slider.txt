  <script>
    const thumbnails = document.querySelectorAll('.thumbnail');
    const imageViewer = document.querySelector('.image-viewer');
    const viewedImage = document.querySelector('.viewed-image');
    const closeBtn = document.querySelector('.close-btn');
    const prevBtn = document.querySelector('.prev-btn');
    const nextBtn = document.querySelector('.next-btn');
    let currentIndex = 0;

    thumbnails.forEach((thumbnail, index) => {
      thumbnail.addEventListener('click', () => {
        currentIndex = index;
        updateViewer();
        imageViewer.style.display = 'block';
      });
    });

    closeBtn.addEventListener('click', () => {
      imageViewer.style.display = 'none';
    });

    prevBtn.addEventListener('click', () => {
      currentIndex = (currentIndex - 1 + thumbnails.length) % thumbnails.length;
      updateViewer();
    });

    nextBtn.addEventListener('click', () => {
      currentIndex = (currentIndex + 1) % thumbnails.length;
      updateViewer();
    });

    function updateViewer() {
      const src = thumbnails[currentIndex].querySelector('img').getAttribute('src');
      viewedImage.src = src;
    };

    document.addEventListener('DOMContentLoaded', function () {
  const prevBtn = document.querySelector('.pev-btn');
  const nextBtn = document.querySelector('.nt-btn');
  
  prevBtn.addEventListener('click', function () {
    $('#carouselExampleControls').carousel('prev');
  });
  
  nextBtn.addEventListener('click', function () {
    $('#carouselExampleControls').carousel('next');
  });
});

  </script>
